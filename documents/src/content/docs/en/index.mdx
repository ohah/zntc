        ```bash
        # install

        # bundle
        zts build src/index.ts --bundle --outdir dist

        # watch mode
        zts build src/index.ts --bundle --watch
        ```
      </TabItem>

      <TabItem label="JS API">
        ```ts
        import { build } from "@zts/core";

        const result = await build({
          entryPoints: ["src/index.ts"],
          bundle: true,
          outdir: "dist",
          target: "es2020",
          platform: "browser",
        });

        console.log(`${result.outputFiles.length} files emitted`);
        ```
      </TabItem>

      <TabItem label="Vite Plugin">

        ```ts
        import { defineConfig } from "vite";

        export default defineConfig({
          plugins: [
            zts({
              jsx: "automatic",
              target: "es2022",
            }),
          ],
        });
        ```
