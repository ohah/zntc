  <Tabs>
    <TabItem label="CLI">
      ```bash
      # install
      npm install -g @zntc/core

      # bundle
      zntc build src/index.ts --bundle --outdir dist

      # watch mode
      zntc build src/index.ts --bundle --watch
      ```
    </TabItem>

    <TabItem label="JS API">
      ```ts
      import { build } from "@zntc/core";

      const result = await build({
        entryPoints: ["src/index.ts"],
        bundle: true,
        outdir: "dist",
        target: "es2020",
        platform: "browser",
      });

      console.log(`${result.outputFiles.length} files emitted`);
      ```
    </TabItem>

    <TabItem label="Vite Plugin">
      ```ts
      // vite.config.ts
      import { defineConfig } from "vite";
      import zntc from "@zntc/vite-plugin";

      export default defineConfig({
        plugins: [
          zntc({
            jsx: "automatic",
            target: "es2022",
          }),
        ],
      });
      ```
    </TabItem>

    <TabItem label="Rspack / Webpack">
      ```js
      // rspack.config.mjs
      export default {
        module: {
          rules: [
            {
              test: /\.(?:tsx?|jsx?)$/,
              exclude: /node_modules/,
              loader: "@zntc/rspack-loader",
              options: {
                transpileOptions: { target: "es2020", jsx: "automatic" },
              },
            },
          ],
        },
      };
      ```
    </TabItem>
  </Tabs>
